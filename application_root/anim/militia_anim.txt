anim Bored
{
  SetSelector
  {
	TechLogic none/Medium Infantry/Heavy Infantry/Champion Infantry
      set hotspot
      {
         version
         {
            Visual Infantry G Militia_BoredA
	connect RightHand Spear hotspot
         }
      }
      set hotspot
      {
         version
         {
            Visual Infantry G Militia_BoredA
	ReplaceTexture Infantry G militia/Infantry G militia medium
	ReplaceTexture Infantry G militia head/Infantry G militia head medium
	connect RightHand Spear hotspot
        }
      }
      set hotspot
      {
         version
         {
            Visual Infantry G Militia_BoredA
	ReplaceTexture Infantry G militia/Infantry G militia heavy
	ReplaceTexture Infantry G militia head/Infantry G militia head heavy
	connect RightHand Spear hotspot

      }
      set hotspot
      {
         version
         {
            Visual Infantry G Militia_BoredA
	ReplaceTexture Infantry G militia/Infantry G militia champion
	ReplaceTexture Infantry G militia head/Infantry G militia head champion
	connect RightHand Spear hotspot
         }
      }
   }
}
