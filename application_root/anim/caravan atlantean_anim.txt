anim idle
{
  SetSelector
  {
    Techlogic None/Coinage
    SetSelector
    {
      InventoryLogic 0 1
      set hotspot
      {
         version
         {
            Visual Trade X Caravan_idlea
     	tag SpecificSoundSet 0.30 false coins checkowner 
         }
      }
      set hotspot
      {
         version
         {
            Visual Trade X Caravan_LoadedIdlea
     	tag SpecificSoundSet 0.30 false coins checkowner 
         }
      }   
    }
    SetSelector
    {
      InventoryLogic 0 1
      set hotspot
      {
         version
         {
            Visual Trade X Caravan_idlea
		replacetexture trade x caravan/trade x caravan coinage
     	tag SpecificSoundSet 0.30 false coins checkowner 
         }
      }
      set hotspot
      {
         version
         {
            Visual Trade X Caravan_LoadedIdlea
		replacetexture trade x caravan/trade x caravan coinage
     	tag SpecificSoundSet 0.30 false coins checkowner 
         }
      }   
    }
  }
}
