//==============================================================================
anim Birth
{
   SetSelector
   {
Techlogic none/Labyrinth of Minos
      set hotspot
      {
         version
         {
            Visual Special G Minotaur_birthA
            connect RightHand axe hotspot
				connect HOTSPOT greekbirthSFX hotspot
         }
      }
      set hotspot
      {
         version
         {
            Visual Special G Minotaur_birthA
replacetexture special g minotaur/special g minotaur upgrade
            connect RightHand axe hotspot
				connect HOTSPOT greekbirthSFX hotspot
         }
      }

   }
}
